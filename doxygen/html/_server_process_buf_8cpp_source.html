89 {

92  std::ptrdiff_t(16 - sizeof(int))));

103 int ServerProcessBuf::sync(){
104  if (this->pbase() != this->pptr()) {
105  std::streamsize size(this->pptr() - this->pbase());
106  std::streamsize done(::write(this->childProcessStdin[PIPE_WRITE], this->outbuf_, size));
107  // The code below assumes that it is success if the stream made
108  // some progress. Depending on the needs it may be more
109  // reasonable to consider it a success only if it managed to
110  // write the entire buffer and, e.g., loop a couple of times
111  // to try achieving this success.
112  if (0 < done) {
113  std::copy(this->pbase() + done, this->pptr(), this->pbase());
114  this->setp(this->pbase(), this->epptr());
115  this->pbump(size - done);
116  }
117  }
118  return this->pptr() != this->epptr()? 0: -1;
119 }
