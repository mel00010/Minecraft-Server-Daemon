104 static inline std::string codePointToUTF8(unsigned int cp) {
105  std::string result;
106 
107  // based on description from http://en.wikipedia.org/wiki/UTF-8
108 
109  if (cp <= 0x7f) {
110  result.resize(1);
111  result[0] = static_cast<char>(cp);
112  } else if (cp <= 0x7FF) {
113  result.resize(2);
114  result[1] = static_cast<char>(0x80 | (0x3f & cp));
115  result[0] = static_cast<char>(0xC0 | (0x1f & (cp >> 6)));
116  } else if (cp <= 0xFFFF) {
117  result.resize(3);
118  result[2] = static_cast<char>(0x80 | (0x3f & cp));
119  result[1] = static_cast<char>(0x80 | (0x3f & (cp >> 6)));
120  result[0] = static_cast<char>(0xE0 | (0xf & (cp >> 12)));
121  } else if (cp <= 0x10FFFF) {
122  result.resize(4);
123  result[3] = static_cast<char>(0x80 | (0x3f & cp));
124  result[2] = static_cast<char>(0x80 | (0x3f & (cp >> 6)));
125  result[1] = static_cast<char>(0x80 | (0x3f & (cp >> 12)));
126  result[0] = static_cast<char>(0xF0 | (0x7 & (cp >> 18)));
127  }
128 
129  return result;
130 }

349 bool Reader::readValue() {
350  // This is a non-reentrant way to support a stackLimit. Terrible!
351  // But this deprecated class has a security problem: Bad input can
352  // cause a seg-fault. This seems like a fair, binary-compatible way
353  // to prevent the problem.
354  if (stackDepth_g >= stackLimit_g) throwRuntimeError("Exceeded stackLimit in readValue().");
355  ++stackDepth_g;
356 
357  Token token;
358  skipCommentTokens(token);
359  bool successful = true;
360 
361  if (collectComments_ && !commentsBefore_.empty()) {
362  currentValue().setComment(commentsBefore_, commentBefore);
363  commentsBefore_ = "";
364  }
365 
366  switch (token.type_) {
367  case tokenObjectBegin:
368  successful = readObject(token);
369  currentValue().setOffsetLimit(current_ - begin_);
370  break;
371  case tokenArrayBegin:
372  successful = readArray(token);
373  currentValue().setOffsetLimit(current_ - begin_);
374  break;
375  case tokenNumber:
376  successful = decodeNumber(token);
377  break;
378  case tokenString:
379  successful = decodeString(token);
380  break;
381  case tokenTrue:
382  {
383  Value v(true);
384  currentValue().swapPayload(v);
385  currentValue().setOffsetStart(token.start_ - begin_);
386  currentValue().setOffsetLimit(token.end_ - begin_);
387  }
388  break;
389  case tokenFalse:
390  {
391  Value v(false);
392  currentValue().swapPayload(v);
393  currentValue().setOffsetStart(token.start_ - begin_);
394  currentValue().setOffsetLimit(token.end_ - begin_);
395  }
396  break;
397  case tokenNull:
398  {
399  Value v;
400  currentValue().swapPayload(v);
401  currentValue().setOffsetStart(token.start_ - begin_);
402  currentValue().setOffsetLimit(token.end_ - begin_);
403  }
404  break;
405  case tokenArraySeparator:
406  case tokenObjectEnd:
407  case tokenArrayEnd:
408  if (features_.allowDroppedNullPlaceholders_) {
409  // "Un-read" the current token and mark the current value as a null
410  // token.
411  current_--;
412  Value v;
413  currentValue().swapPayload(v);
414  currentValue().setOffsetStart(current_ - begin_ - 1);
415  currentValue().setOffsetLimit(current_ - begin_);
416  break;
417  } // Else, fall through...
418  default:
419  currentValue().setOffsetStart(token.start_ - begin_);
420  currentValue().setOffsetLimit(token.end_ - begin_);
421  return addError("Syntax error: value, object or array expected.", token);
422  }
423 
424  if (collectComments_) {
425  lastValueEnd_ = current_;
426  lastValue_ = &currentValue();
427  }
428 
429  --stackDepth_g;
430  return successful;
431 }

443 bool Reader::readToken(Token& token) {
444  skipSpaces();
445  token.start_ = current_;
446  Char c = getNextChar();
447  bool ok = true;
448  switch (c) {
449  case '{':
450  token.type_ = tokenObjectBegin;
451  break;
452  case '}':
453  token.type_ = tokenObjectEnd;
454  break;
455  case '[':
456  token.type_ = tokenArrayBegin;
457  break;
458  case ']':
459  token.type_ = tokenArrayEnd;
460  break;
461  case '"':
462  token.type_ = tokenString;
463  ok = readString();
464  break;
465  case '/':
466  token.type_ = tokenComment;
467  ok = readComment();
468  break;
469  case '0':
470  case '1':
471  case '2':
472  case '3':
473  case '4':
474  case '5':
475  case '6':
476  case '7':
477  case '8':
478  case '9':
479  case '-':
480  token.type_ = tokenNumber;
481  readNumber();
482  break;
483  case 't':
484  token.type_ = tokenTrue;
485  ok = match("rue", 3);
486  break;
487  case 'f':
488  token.type_ = tokenFalse;
489  ok = match("alse", 4);
490  break;
491  case 'n':
492  token.type_ = tokenNull;
493  ok = match("ull", 3);
494  break;
495  case ',':
496  token.type_ = tokenArraySeparator;
497  break;
498  case ':':
499  token.type_ = tokenMemberSeparator;
500  break;
501  case 0:
502  token.type_ = tokenEndOfStream;
503  break;
504  default:
505  ok = false;
506  break;
507  }
508  if (!ok)
509  token.type_ = tokenError;
510  token.end_ = current_;
511  return true;
512 }

648 bool Reader::readObject(Token& tokenStart) {
649  Token tokenName;
650  std::string name;
651  Value init(objectValue);
652  currentValue().swapPayload(init);
653  currentValue().setOffsetStart(tokenStart.start_ - begin_);
654  while (readToken(tokenName)) {
655  bool initialTokenOk = true;
656  while (tokenName.type_ == tokenComment && initialTokenOk)
657  initialTokenOk = readToken(tokenName);
658  if (!initialTokenOk)
659  break;
660  if (tokenName.type_ == tokenObjectEnd && name.empty()) // empty object
661  return true;
662  name = "";

664  if (!decodeString(tokenName, name))
665  return recoverFromError(tokenObjectEnd);
666  } else if (tokenName.type_ == tokenNumber && features_.allowNumericKeys_) {
667  Value numberName;
668  if (!decodeNumber(tokenName, numberName))
669  return recoverFromError(tokenObjectEnd);
670  name = numberName.asString();
671  } else {
672  break;
673  }

676  if (!readToken(colon) || colon.type_ != tokenMemberSeparator) {
677  return addErrorAndRecover(

679  }

685  return recoverFromError(tokenObjectEnd);
686

695  while (comma.type_ == tokenComment && finalizeTokenOk)
696  finalizeTokenOk = readToken(comma);
697  if (comma.type_ == tokenObjectEnd)
698  return true;
699  }
700  return addErrorAndRecover(
701  "Missing '}' or object member name", tokenName, tokenObjectEnd);
702 }

752 bool Reader::decodeNumber(Token& token, Value& decoded) {
753  // Attempts to parse the number as an integer. If the number is
754  // larger than the maximum supported value of an integer then
755  // we decode the number as a double.
756  Location current = token.start_;
757  bool isNegative = *current == '-';
758  if (isNegative)
759  ++current;
760  // TODO: Help the compiler do the div and mod at compile time or get rid of them.
761  Value::LargestUInt maxIntegerValue =
762  isNegative ? Value::LargestUInt(Value::maxLargestInt) + 1
763  : Value::maxLargestUInt;
764  Value::LargestUInt threshold = maxIntegerValue / 10;
765  Value::LargestUInt value = 0;
766  while (current < token.end_) {
767  Char c = *current++;
768  if (c < '0' || c > '9')
769  return decodeDouble(token, decoded);
770  Value::UInt digit(static_cast<Value::UInt>(c - '0'));
771  if (value >= threshold) {
772  // We've hit or exceeded the max value divided by 10 (rounded down). If
773  // a) we've only just touched the limit, b) this is the last digit, and
774  // c) it's small enough to fit in that rounding delta, we're okay.
775  // Otherwise treat this number as a double to avoid overflow.
776  if (value > threshold || current != token.end_ ||
777  digit > maxIntegerValue % 10) {
778  return decodeDouble(token, decoded);
779  }
780  }
781  value = value * 10 + digit;
782  }
783  if (isNegative && value == maxIntegerValue)
784  decoded = Value::minLargestInt;
785  else if (isNegative)
786  decoded = -Value::LargestInt(value);
787  else if (value <= Value::LargestUInt(Value::maxInt))
788  decoded = Value::LargestInt(value);
789  else
790  decoded = value;
791  return true;
792 }

1735 bool OurReader::decodeNumber(Token& token, Value& decoded) {
1736  // Attempts to parse the number as an integer. If the number is
1737  // larger than the maximum supported value of an integer then
1738  // we decode the number as a double.
1739  Location current = token.start_;
1740  bool isNegative = *current == '-';
1741  if (isNegative)
1742  ++current;
1743  // TODO: Help the compiler do the div and mod at compile time or get rid of them.
1744  Value::LargestUInt maxIntegerValue =
1745  isNegative ? Value::LargestUInt(-Value::minLargestInt)
1746  : Value::maxLargestUInt;
1747  Value::LargestUInt threshold = maxIntegerValue / 10;
1748  Value::LargestUInt value = 0;
1749  while (current < token.end_) {
1750  Char c = *current++;
1751  if (c < '0' || c > '9')
1752  return decodeDouble(token, decoded);
1753  Value::UInt digit(static_cast<Value::UInt>(c - '0'));
1754  if (value >= threshold) {
1755  // We've hit or exceeded the max value divided by 10 (rounded down). If
1756  // a) we've only just touched the limit, b) this is the last digit, and
1757  // c) it's small enough to fit in that rounding delta, we're okay.
1758  // Otherwise treat this number as a double to avoid overflow.
1759  if (value > threshold || current != token.end_ ||
1760  digit > maxIntegerValue % 10) {
1761  return decodeDouble(token, decoded);
1762  }
1763  }
1764  value = value * 10 + digit;
1765  }
1766  if (isNegative)
1767  decoded = -Value::LargestInt(value);
1768  else if (value <= Value::LargestUInt(Value::maxInt))
1769  decoded = Value::LargestInt(value);
1770  else
1771  decoded = value;
1772  return true;
1773 }
